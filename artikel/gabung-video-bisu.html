#!/bin/bash
# =========================================
# Script: gabung_video_bisu.sh
# Fungsi: Menggabungkan semua file .mp4 menjadi satu video BISU (tanpa suara)
# Fitur: Deteksi otomatis codec/resolusi â€” re-encode jika tidak seragam
# =========================================

set -e
OUTPUT="${1:-hasil_gabungan_bisu.mp4}"
rm -f daftar.txt temp_*.mp4

FILES=( *.mp4 )
if [ ${#FILES[@]} -eq 0 ]; then
  echo "âŒ Tidak ditemukan file .mp4 di folder ini."
  exit 1
fi

REF="${FILES[0]}"
REF_INFO=$(ffprobe -v error -select_streams v:0 -show_entries stream=codec_name,width,height,r_frame_rate -of csv=p=0 "$REF")

REENCODE_NEEDED=false
for f in "${FILES[@]}"; do
  INFO=$(ffprobe -v error -select_streams v:0 -show_entries stream=codec_name,width,height,r_frame_rate -of csv=p=0 "$f")
  if [ "$INFO" != "$REF_INFO" ]; then
    REENCODE_NEEDED=true
  fi
done

if [ "$REENCODE_NEEDED" = true ]; then
  for f in "${FILES[@]}"; do
    ffmpeg -y -i "$f" -an -c:v libx264 -preset veryfast -crf 23 "temp_$f"
    echo "file 'temp_$f'" >> daftar.txt
  done
else
  for f in "${FILES[@]}"; do
    echo "file '$PWD/$f'" >> daftar.txt
  done
fi

ffmpeg -f concat -safe 0 -i daftar.txt -an -c:v libx264 -preset veryfast -crf 23 "$OUTPUT"
rm -f daftar.txt temp_*.mp4 2>/dev/null || true
echo "âœ¨ Selesai! ğŸ“½ï¸ Video bisu hasil: $OUTPUT"
